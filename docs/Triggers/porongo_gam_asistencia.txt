select personasucursal.id from personasucursal,persona
where personasucursal.id_persona=persona.id
and persona.codigo='0'
)

select PersonaGrupo.id from PersonaGrupo,persona
where PersonaGrupo.id_persona=persona.id
and persona.codigo='0'
)

select personaseccion.id from PersonaSeccion,persona

)
